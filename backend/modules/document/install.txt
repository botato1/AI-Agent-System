# 설치 순서 (Windows - CUDA 12.6)
# 아래 3개를 순서대로 복사 붙여넣기

pip install torch torchvision --index-url https://download.pytorch.org/whl/cu126

pip install paddlepaddle==3.0.0

pip install -r requirements.txt
